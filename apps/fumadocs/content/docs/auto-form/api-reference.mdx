## createAutoForm

Assemble un adaptateur, un registre et un layout en un composant `AutoForm`.

```ts
function createAutoForm(config: CreateAutoFormConfig): React.ComponentType<CreatedAutoFormProps>
```

### CreateAutoFormConfig

```ts
interface CreateAutoFormConfig {
  adapter: FormAdapter
  layout?: Partial<FormLayout>
  registry: FieldRegistry
  resolveSchema?: (rawSchema: unknown, draftHint?: JsonSchemaDraft) => ResolvedSchema
  traverseSchema?: (resolved: ResolvedSchema) => FieldMeta[]
  unionFieldRenderer?: ComponentType<AutoFormFieldProps>
}
```

### CreatedAutoFormProps

```ts
interface CreatedAutoFormProps {
  children?: ReactNode
  className?: string
  defaultValues?: Record<string, unknown>
  onSubmit?: (data: unknown) => void | Promise<void>
  schema: Record<string, unknown>
}
```

## AutoFormBuilder

Render-prop builder pour un contrôle total du rendu.

```tsx
<AutoFormBuilder
  adapter={tanstackAdapter}
  registry={createShadcnRegistry()}
  schema={mySchema}
  onSubmit={(data) => console.log(data)}
>
  {({ fields, form, resolvedSchema }) => (
    <form onSubmit={(e) => { e.preventDefault(); form.handleSubmit() }}>
      {fields.map((field) => (/* ... */))}
    </form>
  )}
</AutoFormBuilder>
```

### Props

```ts
interface AutoFormBuilderProps {
  adapter: FormAdapter
  children: (props: AutoFormBuilderChildrenProps) => ReactNode
  defaultValues?: Record<string, unknown>
  onSubmit?: (data: unknown) => void | Promise<void>
  resolveSchema?: (rawSchema: unknown, draftHint?: JsonSchemaDraft) => ResolvedSchema
  schema: Record<string, unknown>
  traverseSchema?: (resolved: ResolvedSchema) => FieldMeta[]
}
```

### Children props

```ts
interface AutoFormBuilderChildrenProps {
  fields: FieldMeta[]
  form: FormAPI
  resolvedSchema: ResolvedSchema
}
```

## AutoFormField

Composant récursif qui résout un `FieldMeta` en composant de champ.

```tsx
<AutoFormField
  adapter={tanstackAdapter}
  fieldMeta={field}
  form={form}
  registry={createShadcnRegistry()}
  unionFieldRenderer={UnionRenderer}
/>
```

Gère les cas suivants :
- **Primitive** → résout via `registry.resolve(fieldMeta)` et rend le composant
- **Object** → wrapper `layout.ObjectField` avec rendu récursif des enfants
- **Array** → wrapper `layout.ArrayField` avec boutons ajouter/supprimer
- **Union (oneOf/anyOf)** → sélecteur de variante avec rendu conditionnel

## FormAdapter

Interface standardisée pour connecter un form manager.

```ts
interface FormAdapter {
  readonly name: string
  readonly FormProvider: React.ComponentType<FormProviderProps>
  readonly Field: React.ComponentType<FieldProps>
}
```

### FormProviderProps

```ts
interface FormProviderProps {
  children: (form: FormAPI) => React.ReactNode
  defaultValues?: Record<string, unknown>
  onSubmit?: (data: unknown) => void | Promise<void>
  schema?: Record<string, unknown>
  validate?: (values: Record<string, unknown>) => Record<string, FieldError | undefined>
}
```

### FieldProps

```ts
interface FieldProps {
  children: (field: FieldAPI) => React.ReactNode
  name: string
}
```

### FieldAPI

```ts
interface FieldAPI {
  error?: FieldError
  isTouched: boolean
  onBlur: () => void
  onChange: (value: unknown) => void
  value: unknown
}
```

### FormAPI

```ts
interface FormAPI {
  appendFieldValue: (name: string, value: unknown) => void
  handleSubmit: () => void
  isSubmitting: boolean
  removeFieldValue: (name: string, index: number) => void
  reset: () => void
  values: Record<string, unknown>
}
```

## createShadcnRegistry

Crée un `FieldRegistry` préconfiguré avec les composants shadcn/ui.

```ts
function createShadcnRegistry(): FieldRegistry
```

Sélecteurs enregistrés :

| Type | Sélecteur | Composant |
|---|---|---|
| `string` enum | `{ type: "string", kind: "enum" }` | ShadcnEnumField |
| `string` textarea | `{ type: "string", widget: "textarea" }` | ShadcnTextareaField |
| `string` password | `{ type: "string", widget: "password" }` | ShadcnPasswordField |
| `string` date | `{ type: "string", format: "date" }` | ShadcnDateField |
| `string` default | `{ type: "string", kind: "primitive" }` | ShadcnTextField |
| `number` | `{ type: "number" }` | ShadcnNumberField |
| `integer` | `{ type: "integer" }` | ShadcnNumberField |
| `boolean` switch | `{ type: "boolean", widget: "switch" }` | ShadcnSwitchField |
| `boolean` default | `{ type: "boolean", kind: "primitive" }` | ShadcnBooleanField |
| fallback | — | ShadcnTextField |

## FormLayout

Interface pour la stratégie de rendu des conteneurs.

```ts
interface FormLayout {
  ArrayField: ComponentType<ArrayFieldProps>
  CompositionsField: ComponentType<CompositionsFieldProps>
  FieldDescription: ComponentType<{ children: ReactNode }>
  FieldGroup: ComponentType<{ children: ReactNode }>
  FieldLegend: ComponentType<{ children: ReactNode; className?: string }>
  FieldSet: ComponentType<{ children: ReactNode; className?: string }>
  ObjectField: ComponentType<ObjectFieldProps>
  SubmitButton: ComponentType<{ children?: ReactNode; disabled?: boolean; isSubmitting?: boolean }>
}
```

Layout par défaut : `shadcnLayout` (basé sur les composants `Field` de shadcn/ui).

## Composants de champ

Tous les composants sont exportés depuis `@code2-base-ui/auto-form-builder/fields` :

```ts
import {
  ShadcnBooleanField,
  ShadcnDateField,
  ShadcnEnumField,
  ShadcnNumberField,
  ShadcnPasswordField,
  ShadcnSwitchField,
  ShadcnTextareaField,
  ShadcnTextField,
} from "@code2-base-ui/auto-form-builder/fields"
```

Ils acceptent tous les props `FieldComponentProps` :

```ts
interface FieldComponentProps {
  className?: string
  disabled?: boolean
  error?: string | { message: string; type?: string; path?: string[] }
  field?: FieldMeta
  id?: string
  label?: string
  onChange?: (value: unknown) => void
  placeholder?: string
  value?: unknown
}
```
